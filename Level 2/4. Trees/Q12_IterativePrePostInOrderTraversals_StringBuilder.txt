time O(n)
space O(n)  (stack + output strings)

Why StringBuilder here:
- Tree traversal itself is O(n) because each node is processed a constant number of times.
- With normal String (`pre += ...`), Java creates a new String every time and copies old content.
- Repeating that copy many times can make total cost O(n^2).
- `StringBuilder` appends in-place (amortized O(1) per append), so total output-building cost stays O(n).
- Therefore full solution remains O(n) with StringBuilder.

class Solution {
  public static class TPair {
    Node node;
    int state;

    TPair(Node node, int state) {
      this.node = node;
      this.state = state;
    }
  }

  public static void iterativePrePostInTraversal(Node node) {
    Stack<TPair> stack = new Stack<>();
    TPair rootp = new TPair(node, 1);
    stack.push(rootp);

    StringBuilder pre = new StringBuilder();
    StringBuilder in = new StringBuilder();
    StringBuilder post = new StringBuilder();

    while (stack.size() > 0) {
      TPair top = stack.peek();

      if (top.state == 1) {
        // pre = pre, left
        pre.append(top.node.data).append(" ");
        if (top.node.left != null) {
          TPair leftp = new TPair(top.node.left, 1);
          stack.push(leftp);
        }
      } else if (top.state == 2) {
        // in = in, right
        in.append(top.node.data).append(" ");
        if (top.node.right != null) {
          TPair rightp = new TPair(top.node.right, 1);
          stack.push(rightp);
        }
      } else {
        // post = post, pop
        post.append(top.node.data).append(" ");
        stack.pop();
      }

      top.state++;
    }

    System.out.println(pre.toString());
    System.out.println(in.toString());
    System.out.println(post.toString());
  }
}
