Time O(n log n)
Space O(n)

class Solution {
  public int minRefuelStops(int target, int startFuel, int[][] stations) {
    PriorityQueue<Integer> queue = new PriorityQueue<>(Collections.reverseOrder());

    int range = startFuel;
    int i = 0;
    int pumps = 0;

    // Keep refueling until the target becomes reachable
    while (range < target) {

      // Add all stations that are reachable with the current range
      while (i < stations.length) {
        if (stations[i][0] <= range) {
          queue.add(stations[i][1]);
        } else {
          break;
        }
        i++;
      }

      // No reachable station left to refuel from, so target is impossible
      if (queue.size() == 0) {
        return -1;
      }

      // Refuel from the reachable station having the maximum fuel
      range += queue.remove();

      // Count this refueling stop
      pumps++;
    }

    return pumps;
  }
}
