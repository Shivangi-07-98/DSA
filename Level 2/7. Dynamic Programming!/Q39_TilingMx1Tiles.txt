class Solution {

  static int MOD = 1000000007;

  public int countWays(int n, int m) {
    long[] dp = new long[n + 1];

    for (int i = 1; i < dp.length; i++) {
      if (i < m) {
        dp[i] = 1;
      } else if (i == m) {
        dp[i] = 2;
      } else {
        dp[i] = (dp[i - 1] + dp[i - m]) % MOD;
      }
    }

    return (int) dp[n];
  }
}
