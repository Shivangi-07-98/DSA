Time O(n log n)
Space O(1)

class Solution {
  boolean twoSum(int arr[], int target) {
    Arrays.sort(arr);

    int l = 0;
    int r = arr.length - 1;

    while (l < r) {
      if (arr[l] + arr[r] < target) {
        l++;
      } else if (arr[l] + arr[r] > target) {
        r--;
      } else {
        return true;
      }
    }

    return false;
  }
}
