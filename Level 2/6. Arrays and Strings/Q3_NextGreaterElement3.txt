class Solution {
  public int nextGreaterElement(int n) {
    char[] arr = (n + "").toCharArray();

    int i = arr.length - 1;
    while (i > 0) {
      if (arr[i - 1] >= arr[i]) {
        i--;
      } else {
        break;
      }
    }

    if (i == 0) {
      return -1;
    }

    int idx1 = i - 1;

    int j = arr.length - 1;
    while (j > idx1) {
      if (arr[j] > arr[idx1]) {
        break;
      }
      j--;
    }

    // swap
    swap(arr, idx1, j);

    // reverse from end to i
    int left = i;
    int right = arr.length - 1;
    while (left < right) {
      swap(arr, left, right);
      left++;
      right--;
    }

    // char arr se wapas string and then single integer
    String res = new String(arr); // char arr pass krdo string bn jayegi
    // int val = Integer.parseInt(res); // int max is only about 2.147 × 10^9
    long val = Long.parseLong(res);
    return (val > Integer.MAX_VALUE ? -1 : (int) val); // long nhi return kr skte int mai typecast
  }

  void swap(char[] arr, int i, int j) {
    char temp = arr[i];
    arr[i] = arr[j];
    arr[j] = temp;
  }
  
}

algorithm - inhi digits ko rearrange krna hai aur usse bada digit bnana hai, for eg 8 bade digits bne rearrange karke toh un bado mai sbse chhota wala ans hai

last se badhte jayega jbtk usse bada number mil rha, jaise hi usse chhota milega uss number ko yaad rakhenge fir uss number se last tk ka koi usse just bada ho toh dono ko swap kr denge, and last ke saare numbers jo decreasing order mai hai usko sort krke increasing order mai kar denge
