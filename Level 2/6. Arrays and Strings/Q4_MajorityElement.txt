time O(n)
space O(1)

class Solution {
  public int majorityElement(int[] nums) {
    int count = 0;
    Integer me = null; // primitive int cannot store null
    // int me = 0; // we could have used this here

    for (int i = 0; i < nums.length; i++) {
      if (count == 0) {
        me = nums[i];
        count = 1;
      } else if (nums[i] == me) {
        count++;
      } else {
        count--;
      }
    }

    return me;
  }
}
