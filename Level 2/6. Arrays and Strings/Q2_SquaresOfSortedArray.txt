time O(n)
space O(n)

class Solution {
  public int[] sortedSquares(int[] nums) {
    int[] res = new int[nums.length];

    int l = 0;
    int r = nums.length - 1;
    int k = res.length - 1;
    while (k >= 0) {
      if (Math.abs(nums[l]) > Math.abs(nums[r])) {
        res[k] = nums[l] * nums[l];
        l++;
      } else {
        res[k] = nums[r] * nums[r];
        r--;
      }

      k--;
    }

    return res;
  }
}

algorithm - array sorted hai agar sb positive hota toh last wale ka hi bda aata, but negatives hai toh left right comparison hoga, jiska bda uska ans mai last daal kar sirf uska pointer badha denge
