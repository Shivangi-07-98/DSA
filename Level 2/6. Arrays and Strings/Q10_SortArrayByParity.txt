time O(n)
space O(1)

class Solution {
  public int[] sortArrayByParity(int[] arr) {
    int i = 0;
    int j = 0;
    while (i < arr.length) {
      if (arr[i] % 2 == 0) {
        swap(arr, i, j);
        i++;
        j++;
      } else {
        i++;
      }
    }

    return arr;
  }

  public void swap(int[] arr, int i, int j) {
    int temp = arr[j];
    arr[j] = arr[i];
    arr[i] = temp;
  }
}
