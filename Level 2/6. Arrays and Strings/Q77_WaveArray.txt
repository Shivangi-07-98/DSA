class Solution {
  static void convertToWave(int n, int[] a) {
    Arrays.sort(a);

    for (int i = 0; i < a.length; i += 2) {
      if (i < a.length - 1) {
        int temp = a[i + 1];
        a[i + 1] = a[i];
        a[i] = temp;
      }
    }
  }
}
