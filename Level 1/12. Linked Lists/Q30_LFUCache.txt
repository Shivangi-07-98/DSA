class LFUCache {

  class Node {
    int val;
    int key;
    int freq;
    Node next;
    Node prev;

    Node(int k, int v) {
      key = k;
      val = v;
      freq = 1;
    }
  }

  class LL {
    Node head;
    Node tail;
    int size;

    LL() {
      head = new Node(0, 0);
      tail = new Node(0, 0);
      head.next = tail;
      tail.prev = head;
      size = 0;
    }
  }

  private void remove(LL list, Node node) {
    Node m1 = node.prev;
    Node p1 = node.next;

    m1.next = p1;
    p1.prev = m1;

    list.size--;
  }

  private void addFirst(LL list, Node node) {
    Node m1 = list.head;
    Node p1 = list.head.next;

    node.prev = m1;
    node.next = p1;
    m1.next = node;
    p1.prev = node;

    list.size++;
  }

  int cap;
  int min;
  HashMap<Integer, Node> nodemap;
  HashMap<Integer, LL> listmap;

  public LFUCache(int capacity) {
    cap = capacity;
    min = 1;
    nodemap = new HashMap<>();
    listmap = new HashMap<>();
  }


  public int get(int key) {
    if (nodemap.containsKey(key) == false) {
      return -1;
    } else {
      Node node = nodemap.get(key);
      LL list1 = listmap.get(node.freq);

      remove(list1, node);
      if (list1.size == 0 && min == node.freq) {
        min++;
      }

      node.freq++;
      LL list2;
      if (listmap.containsKey(node.freq) == false) {
        list2 = new LL();
        listmap.put(node.freq, list2);
      } else {
        list2 = listmap.get(node.freq);
      }
      addFirst(list2, node);

      return node.val;
    }
  }

  public void put(int key, int value) {
    if (cap == 0) {
      return;
    }

    if (nodemap.containsKey(key)) {
      Node node = nodemap.get(key);
      node.val = value;

      LL list1 = listmap.get(node.freq);
      remove(list1, node);
      if (list1.size == 0 && min == node.freq) {
        min++;
      }

      node.freq++;
      LL list2;
      if (listmap.containsKey(node.freq) == false) {
        list2 = new LL();
        listmap.put(node.freq, list2);
      } else {
        list2 = listmap.get(node.freq);
      }
      addFirst(list2, node);
      return;
    }

    if (nodemap.size() == cap) {
      LL list1 = listmap.get(min);
      Node node = list1.tail.prev;

      remove(list1, node);
      nodemap.remove(node.key);
      if (list1.size == 0) {
        listmap.remove(min);
      }
    }

    Node node = new Node(key, value);
    
    min = 1;
    nodemap.put(key, node);
    LL list1;
    if (listmap.containsKey(node.freq) == false) {
      list1 = new LL();
      listmap.put(node.freq, list1);
    } else {
      list1 = listmap.get(node.freq);
    }
    addFirst(list1, node);
  }
  
}
