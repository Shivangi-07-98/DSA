class GfG {
  public static Node quickSort(Node node) {
    // Your code here
    Node start = node;
    Node end = node;
    while (end.next != null) {
      end = end.next;
    }
    helper(start, end);
    return start;
  }

  static void helper(Node start, Node end) {
    if (start == end || start == null || start == end.next) {
      return;
    }

    Node smallerEnd = partition(start, end);
    if (smallerEnd == null) {
      helper(start.next, end);
    } else {
      helper(start, smallerEnd);
      helper(smallerEnd.next.next, end);
    }
  }

  static Node partition(Node start, Node end) {
    int pivot = end.data;

    Node prev = null;
    Node curr = start;
    while (start != end) {
      if (start.data <= pivot) {
        prev = curr;
        int temp = curr.data;
        curr.data = start.data;
        start.data = temp;
        curr = curr.next;
      }
      start = start.next;
    }

    int temp = curr.data;
    curr.data = start.data;
    start.data = temp;

    return prev;
  }
}
