class Solution {
  public void reorderList(ListNode head) {
    if (head == null || head.next == null) {
      return;
    }

    ListNode slow = head;
    ListNode fast = head;

    while (fast.next != null && fast.next.next != null) {
      slow = slow.next;
      fast = fast.next.next;
    }

    ListNode second = reverse(slow.next);
    slow.next = null;

    ListNode first = head;
    while (second != null) {
      ListNode fNext = first.next;
      ListNode sNext = second.next;

      first.next = second;
      second.next = fNext;

      first = fNext;
      second = sNext;
    }
  }

  private ListNode reverse(ListNode head) {
    ListNode prev = null;
    ListNode curr = head;

    while (curr != null) {
      ListNode next = curr.next;
      curr.next = prev;
      prev = curr;
      curr = next;
    }

    return prev;
  }
  
}
